\unexpanded\def\VAR#1%
  {\def\currentVAR{#1}
   \dosingleargument\doVAR}

\def\doVAR[#1]#2(#3)% #2 = space
 {\doifelsenothing{#1}
  {\currentVAR}
  {\currentVAR_{\NOfAgents#1::\@nil}}%
  \doifsomething{#3}{\TIME#3::\@nil}}

\def\NOfAgents#1:#2:#3\@nil%
  {\doifelsenothing{#2}
   {\doNOfAgents#1,,\@nil}
   {[#1\COLON#2]}}

\def\doNOfAgents#1,#2,#3\@nil%
  {\doifelsenothing{#2}
    {#1}
    {[#1,#2]}}

\def\TIME#1:#2:#3\@nil%
  {(\doifelsenothing{#2}
   {#1}
   {#1\COLON#2})}
